#will set FEC 2:4
echo "2:4" >/tmp/FEC.wfb
#will set FEC 8:10 and MCS index=3
echo "8:10:3" >/tmp/FEC.wfb

#will change only MCS index
echo "0:0:4" >/tmp/FEC.wfb

File is checked after every FEC block is sent. In a scanario with 8:12 FEC this means after every 12-th packet.
If the rate is 1200 packets per second, typical for ~9Mbit/s video, this means refresh time is 10ms.
